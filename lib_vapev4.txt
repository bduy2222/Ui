--// UTILITIES
local function Tween(instance, duration, properties, style, dir)
    local tweenInfo = TweenInfo.new(duration, style or Enum.EasingStyle.Quart, dir or Enum.EasingDirection.Out)
    local tween = TweenService:Create(instance, tweenInfo, properties)
    tween:Play()
    return tween
end

local function MakeDraggable(topbar, object)
    local dragging, dragInput, dragStart, startPos
    topbar.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = object.Position
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement then
            dragInput = input
        end
    end)
    RunService.RenderStepped:Connect(function()
        if dragging and dragInput then
            local delta = dragInput.Position - dragStart
            Tween(object, 0.08, {Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)}, Enum.EasingStyle.Linear)
        end
    end)
    topbar.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
            dragInput = nil
        end
    end)
end

local function CreateRipple(button)
    button.ClipsDescendants = true
    button.MouseButton1Down:Connect(function()
        local mouse = UserInputService:GetMouseLocation()
        local relativePos = mouse - button.AbsolutePosition
        
        local circle = Instance.new("ImageLabel")
        circle.AnchorPoint = Vector2.new(0.5, 0.5)
        circle.Position = UDim2.new(0, relativePos.X, 0, relativePos.Y - 36)
        circle.Size = UDim2.new(0, 0, 0, 0)
        circle.BackgroundTransparency = 1
        circle.Image = Library.Theme.CircleAsset
        circle.ImageColor3 = Library.Theme.Accent
        circle.ImageTransparency = 0.5
        circle.ZIndex = button.ZIndex + 1
        circle.Parent = button

        local maxSize = math.max(button.AbsoluteSize.X, button.AbsoluteSize.Y) * 2.5
        Tween(circle, 0.6, {Size = UDim2.new(0, maxSize, 0, maxSize), ImageTransparency = 1}, Enum.EasingStyle.Exponential)
        task.delay(0.6, function() circle:Destroy() end)
    end)
end

local function AnimateGradient(gradient)
    local connection = RunService.Heartbeat:Connect(function()
        gradient.Rotation = (gradient.Rotation + 1) % 360
    end)
    table.insert(Library.Connections, connection)
end

local function CreateMagicParticles(parent)
    local connection = RunService.Heartbeat:Connect(function()
        if math.random() > 0.95 then
            local particle = Instance.new("ImageLabel")
            particle.Size = UDim2.new(0, math.random(2, 6), 0, math.random(2, 6))
            particle.Position = UDim2.new(math.random(), 0, 1.1, 0)
            particle.BackgroundTransparency = 1
            particle.Image = Library.Theme.CircleAsset
            particle.ImageColor3 = math.random() > 0.5 and Library.Theme.Accent or Library.Theme.AccentSecondary
            particle.ImageTransparency = 0.4
            particle.ZIndex = parent.ZIndex - 1
            particle.Parent = parent

            local duration = math.random(3, 7)
            Tween(particle, duration, {
                Position = UDim2.new(particle.Position.X.Scale + (math.random(-10, 10)/100), 0, -0.1, 0),
                ImageTransparency = 1,
                Size = UDim2.new(0, 0, 0, 0)
            }, Enum.EasingStyle.Sine, Enum.EasingDirection.Out)

            task.delay(duration, function() particle:Destroy() end)
        end
    end)
    table.insert(Library.Connections, connection)
end

--// CONFIG SYSTEM
function Library:SaveConfig()
    if not isfolder or not writefile then return end
    if not isfolder(Library.Config.Folder) then makefolder(Library.Config.Folder) end
    writefile(Library.Config.Folder .. "/" .. Library.Config.File, HttpService:JSONEncode(Library.Flags))
    Library:Notify({Title = "Config Saved", Content = "Successfully saved configuration.", Duration = 3})
end

function Library:LoadConfig()
    if not isfolder or not readfile then return end
    local path = Library.Config.Folder .. "/" .. Library.Config.File
    if isfile(path) then
        local success, data = pcall(function() return HttpService:JSONDecode(readfile(path)) end)
        if success and type(data) == "table" then
            for key, value in pairs(data) do
                if Library.Flags[key] ~= nil then Library.Flags[key] = value end
            end
            Library:Notify({Title = "Config Loaded", Content = "Configuration loaded successfully.", Duration = 3})
        end
    end
end

                BarBack.Size = UDim2.new(1, -20, 0, 4)
                BarBack.Position = UDim2.new(0, 10, 0, 32)
                BarBack.BackgroundColor3 = Library.Theme.Outline
                BarBack.Text = ""
                BarBack.AutoButtonColor = false
                Instance.new("UICorner", BarBack)

                local Fill = Instance.new("Frame", BarBack)
                Fill.Size = UDim2.new((default - min) / (max - min), 0, 1, 0)
                Fill.BackgroundColor3 = Library.Theme.Accent
                Instance.new("UICorner", Fill)
                
                local FillGrad = Instance.new("UIGradient", Fill)
                FillGrad.Color = ColorSequence.new({
                    ColorSequenceKeypoint.new(0, Library.Theme.Accent),
                    ColorSequenceKeypoint.new(1, Library.Theme.AccentSecondary)
                })

                local Thumb = Instance.new("Frame", BarBack)
                Thumb.Size = UDim2.new(0, 10, 0, 10)
                Thumb.Position = UDim2.new((default - min) / (max - min), -5, 0.5, -5)
                Thumb.BackgroundColor3 = Color3.new(1,1,1)
                Instance.new("UICorner", Thumb).CornerRadius = UDim.new(1, 0)
                
                local ThumbGlow = Instance.new("ImageLabel", Thumb)
                ThumbGlow.Size = UDim2.new(1, 16, 1, 16)
                ThumbGlow.Position = UDim2.new(0, -8, 0, -8)
                ThumbGlow.BackgroundTransparency = 1
                ThumbGlow.Image = Library.Theme.GlowAsset
                ThumbGlow.ImageColor3 = Library.Theme.Accent
                ThumbGlow.ImageTransparency = 0.4
                ThumbGlow.ZIndex = 0

                local function UpdateSlider(percent)
                    local value = math.floor(min + ((max - min) * percent))
                    Library.Flags[flag] = value
                    ValueBox.Text = tostring(value)
                    Tween(Fill, 0.1, {Size = UDim2.new(percent, 0, 1, 0)}, Enum.EasingStyle.Linear)
                    Tween(Thumb, 0.1, {Position = UDim2.new(percent, -5, 0.5, -5)}, Enum.EasingStyle.Linear)
                    if opt.Callback then task.spawn(opt.Callback, value) end
                end

                local dragging = false
                BarBack.InputBegan:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 then
                        dragging = true
                        local pct = math.clamp((input.Position.X - BarBack.AbsolutePosition.X) / BarBack.AbsoluteSize.X, 0, 1)
                        UpdateSlider(pct)
                    end
                end)
                UserInputService.InputChanged:Connect(function(input)
                    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
                        local pct = math.clamp((input.Position.X - BarBack.AbsolutePosition.X) / BarBack.AbsoluteSize.X, 0, 1)
                        UpdateSlider(pct)
                    end
                end)
                UserInputService.InputEnded:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end
                end)
            end

            --// DROPDOWN
            function SectionAPI:AddDropdown(opt)
                local flag = opt.Flag or opt.Name
                local options = opt.Options or {}
                local default = opt.Default or options[1]
                Library.Flags[flag] = default

                local DropFrame = Instance.new("TextButton", SecContent)
                DropFrame.Size = UDim2.new(1, 0, 0, 36)
                DropFrame.BackgroundColor3 = Library.Theme.Hover
                DropFrame.BackgroundTransparency = 0.5
                DropFrame.Text = ""
                DropFrame.AutoButtonColor = false
                Instance.new("UICorner", DropFrame).CornerRadius = UDim.new(0, 6)

                local Title = Instance.new("TextLabel", DropFrame)
                Title.Size = UDim2.new(1, -40, 1, 0)
                Title.Position = UDim2.new(0, 10, 0, 0)
                Title.Text = opt.Name .. " : " .. tostring(default)
                Title.TextColor3 = Library.Theme.Text
                Title.Font = Enum.Font.GothamMedium
                Title.TextSize = 12
                Title.TextXAlignment = Enum.TextXAlignment.Left
                Title.BackgroundTransparency = 1

                local Icon = Instance.new("TextLabel", DropFrame)
                Icon.Size = UDim2.new(0, 20, 1, 0)
                Icon.Position = UDim2.new(1, -25, 0, 0)
                Icon.Text = "+"
                Icon.TextColor3 = Library.Theme.Accent
                Icon.Font = Enum.Font.GothamBold
                Icon.TextSize = 14
                Icon.BackgroundTransparency = 1

                local DropContainer = Instance.new("Frame", DropFrame)
                DropContainer.Size = UDim2.new(1, 0, 0, 0)
                DropContainer.Position = UDim2.new(0, 0, 0, 36)
                DropContainer.BackgroundColor3 = Library.Theme.Container
                DropContainer.BackgroundTransparency = 0.1
                DropContainer.ClipsDescendants = true
                DropContainer.ZIndex = 5
                Instance.new("UICorner", DropContainer).CornerRadius = UDim.new(0, 6)
                
                local DropLayout = Instance.new("UIListLayout", DropContainer)
                DropLayout.Padding = UDim.new(0, 2)

                local isDropped = false
                DropFrame.MouseButton1Click:Connect(function()
                    isDropped = not isDropped
                    Tween(Icon, 0.3, {Rotation = isDropped and 45 or 0})
                    Tween(DropContainer, 0.4, {Size = UDim2.new(1, 0, 0, isDropped and DropLayout.AbsoluteContentSize.Y + 4 or 0)}, Enum.EasingStyle.Exponential)
                    Tween(DropFrame, 0.4, {Size = UDim2.new(1, 0, 0, isDropped and DropLayout.AbsoluteContentSize.Y + 40 or 36)}, Enum.EasingStyle.Exponential)
                end)

                local function Populate(list)
                    for _, v in pairs(DropContainer:GetChildren()) do
                        if v:IsA("TextButton") then v:Destroy() end
                    end
                    for _, option in pairs(list) do
                        local Item = Instance.new("TextButton", DropContainer)
                        Item.Size = UDim2.new(1, 0, 0, 26)
                        Item.BackgroundTransparency = 1
                        Item.Text = option
                        Item.TextColor3 = Library.Theme.DarkText
                        Item.Font = Enum.Font.Gotham
                        Item.TextSize = 11
                        Item.ZIndex = 6
                        
                        Item.MouseEnter:Connect(function() Tween(Item, 0.2, {TextColor3 = Library.Theme.Accent}) end)
                        Item.MouseLeave:Connect(function() Tween(Item, 0.2, {TextColor3 = Library.Theme.DarkText}) end)
                        
                        Item.MouseButton1Click:Connect(function()
                            Library.Flags[flag] = option
                            Title.Text = opt.Name .. " : " .. tostring(option)
                            isDropped = false
                            Tween(Icon, 0.3, {Rotation = 0})
                            Tween(DropContainer, 0.4, {Size = UDim2.new(1, 0, 0, 0)}, Enum.EasingStyle.Exponential)
                            Tween(DropFrame, 0.4, {Size = UDim2.new(1, 0, 0, 36)}, Enum.EasingStyle.Exponential)
                            if opt.Callback then task.spawn(opt.Callback, option) end
                        end)
                    end
                end
                Populate(options)
            end

            return SectionAPI
        end

        if not WindowAPI.CurrentTab then
            task.spawn(function() TabBtn.MouseButton1Click:Fire() end)
        end

        return TabAPI
    end

    return WindowAPI
end

return Library
